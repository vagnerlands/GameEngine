#version 330 core
out vec4 FragColor;

in VS_OUT {
    vec3 FragPos;
    vec2 TexCoords;
} fs_in;

uniform sampler2D diffuseMap;
uniform vec4 color;

void main()
{           
	// Activate to DEBUG
    vec4 base = texture(diffuseMap, fs_in.TexCoords);
    if ((base.r < 0.01) && (base.g < 0.01) && (base.b < 0.01))
        base.a = 0.25;
    if ((base.r > 0.9) && (base.b > 0.9))
        base = color;
    FragColor = base;
}
